#include<stdio.h>
void main()
{
    int n,r,rev=0,num;
    printf("Enter n value:");
    scanf("%d",&n);
    num=n;
    while(n!=0)
    {
       r=n%10;
       rev=rev*10+r;
       n=n/10;
    }
       if(num==rev)
       {
           printf("Given number is palindrome");
       }
       else
       {
           printf("Given number is not palindrome");
       }
}
