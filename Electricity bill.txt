    }
    else if(eu>50.0 && eu<=150.0)
    {
        eb=50*0.50+(eu-50)*0.75;

    }
    else if(eu>150.0 && eu<=250.0)
    {
        eb=50*0.50+100*0.75+(eu-150)*1.20;

    }
    else

}
