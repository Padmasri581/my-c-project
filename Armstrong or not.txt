#include<stdio.h>
void main()
{
    int num,rem,total=0,temp;
    printf("enter the number:");
    scanf("%d",&num);
    temp=num;
    while(num>0)
    {
       rem=num%10;
       total=total+rem*rem*rem;
       num=num/10;
    }  
    if(temp==total)
    {
       printf("The number is armstrong");
    }
    else
    {
       printf("The number is not a armstrong");
    }
}
