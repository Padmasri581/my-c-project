#include<stdio.h>
void main()
{ 
    int a,b,c;
    float d;
    printf("enter the values of a,b,c:");
    scanf("%d%d%d",&a,&b,&c);
    d=(b*b)-(4*a*c);
    if(d>0)
    { 
        printf("roots are distinct");
    }
    else if(d=0)
    {
        printf("roots are real");
    }
    else
    {
        printf("roots are complex");
    }
}
